    public getClips(searchText: string) {
        let path = 'rest/restservices/MaterialDistributionSearch';
        let inmediaSession = this.synAuthService.inmediaSession;
        let sessionSign = this.synAuthService.sasSessionSign(path,
                          inmediaSession['private_key'], inmediaSession['session_key']);
        return this.http.get(`${mainUrl}${path}?`
                             + `session_signature=${sessionSign}`
                             + `&ASearchText='${searchText}'`
                             + `&AMode=1&AUserName=''`,
                            { withCredentials: false })
                                .toPromise();
      }
